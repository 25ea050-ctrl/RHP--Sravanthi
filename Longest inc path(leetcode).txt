class Solution {
public:
    const int diff[4][2]={{0,1},{0,-1},{-1,0},{1,0}};
    int dfs(vector<vector<int>>& matrix, vector<vector<int>>& dp, int R,int C ,int r,int c){
        if(dp[r][c]!=0){
            return dp[r][c];
        }

        int adjmax=0;
        for(int i=0;i<4;i++){
            int ar=r+diff[i][0] , ac = c+diff[i][1];
            if(ar>=0 && ar<R && ac>=0 && ac<C && matrix[ar][ac]>matrix[r][c]){
                adjmax=max(adjmax,dfs(matrix,dp,R,C,ar,ac));
            }
        }
        dp[r][c]=1+adjmax;
        return 1+adjmax;
    }

int longestIncreasingPath(vector<vector<int>>& matrix) {
        int R = matrix.size(), C=matrix[0].size(); 
        vector<vector<int>> dp(R,vector<int> (C,0));
        int maxlen=1;
        for(int r=0;r<R;r++){
            for(int c=0;c<C;c++){
                if(dp[r][c]==0){
                    maxlen=max(maxlen,dfs(matrix,dp,R,C,r,c));
                }
            }
        }
    return maxlen;
        
    }
};
